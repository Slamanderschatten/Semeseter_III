12class SimpleCubeCR_OC: public SimpleCube {
13
14public:
15 SimpleCubeCR_OC(GLfloat pSize): SimpleCube(pSize) {
16 glFunctions->glGenQueries(1, &mQuery);
17 glFunctions->glGenQueries(1, &mQueryCR);
18 mSamplesPassed = GL_TRUE;
19 mQueryReady = false;
20 mDelayed = 0;
21 }
22
23 void render() override {
24
25 if (mWithQuery) //Hüllobjekt nur auswerten, wenn Queries aktiv
26 {
27 if (!mConditional) //Abfrage mit Queries ohne Conditional Rendering
28 {
29 if (mDelayed == 0) //neue Query auswerten
30 {
31 /*Begin Query*/
32 glFunctions->glDepthMask(GL_FALSE);
33 glFunctions->glColorMask(GL_FALSE, GL_FALSE, GL_FALSE, GL_FALSE);
34
35 glFunctions->glBeginQuery(GL_ANY_SAMPLES_PASSED, mQuery);
36 SimpleCube::render();
37 glFunctions->glEndQuery(GL_ANY_SAMPLES_PASSED);
38
39 glFunctions->glColorMask(GL_TRUE, GL_TRUE, GL_TRUE, GL_TRUE);
40 glFunctions->glDepthMask(GL_TRUE);
41 /*End Query*/
42 mSamplesPassed = true;
43 }
44 //status abfragen, non-blocking, wenn das Ergebnis nicht da ist, wird nichts in mSamplesPassed geschrieben
45 glFunctions->glGetQueryObjectuiv(mQuery, GL_QUERY_RESULT_NO_WAIT, &mSamplesPassed);
46
47 if (++mDelayed > cNrOfDelayFrames) //Wenn genug Frames vorbei sind -> neue Query starten
48 mDelayed = 0;
49
50 }
51 else //Rendern mit Query und Conditional Rendering
52 {
53 glFunctions->glDepthMask(GL_FALSE);
54 glFunctions->glColorMask(GL_FALSE, GL_FALSE, GL_FALSE, GL_FALSE);
55
56 //Test, ob beim Rendern der Bounding Box Fragments den z-Test bestehen
57 //Auswertung dann beim Rendern des eigentlichen Objekts (s.u.)
58 glFunctions->glBeginQuery(GL_ANY_SAMPLES_PASSED, mQueryCR);
59 SimpleCube::render();
60 glFunctions->glEndQuery(GL_ANY_SAMPLES_PASSED);
61 }
62 }
63 }
64
65 virtual ~SimpleCubeCR_OC() {
66 if (mQuery)
67 glFunctions->glDeleteQueries(1, &mQuery);
68 if (mQueryCR)
69 glFunctions->glDeleteQueries(1, &mQueryCR);
70 }

76 inline bool getConditional() {return(mConditional);}
77 inline void setDoQuery(bool pDoQuery) {mWithQuery = pDoQuery;}
78 inline bool getDoQuery() {return(mWithQuery);}
79 inline GLuint getQueryCR() const {return mQueryCR;}
80

82protected:

84 GLuint mQuery, mQueryCR;

88 unsigned int mDelayed;

91};
